代码

<div id="result">暂无消息</div>

<script src="../dist/cross-window-emitter.min.js"></script>
<script>
  var resultEl = document.getElementById("result");
  var stopBtnEl = document.getElementById("stopBtn");

  function onMessage(value) {
    resultEl.innerHTML = "接受消息：" + value;
  }
  crossWindowEmitter.on('msg', onMessage);

  stopBtnEl.onclick = function () {
    crossWindowEmitter.off('msg');
    resultEl.innerHTML = "已停止监听";
  }
</script>
